  (defun what-line ()
    "Print the current line number (in the buffer) of point."
    (interactive)
    (save-restriction
      (widen)
      (save-excursion
        (beginning-of-line)
        (message "Line %d"
                 (1+ (count-lines 1 (point)))))))

  (defun md-first-sixty-lines ()
    (interactive)
    (save-restriction
      (widen)
      (save-excursion
        (goto-char (point-min))
        (forward-line 60)
        (message "%s"
                 (buffer-substring-no-properties (point-min) (point))))))
